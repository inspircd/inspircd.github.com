183 inline ssize_t writev(int fd, const WindowsIOVec* iov, int count)
184 {
185  DWORD sent;
186  int ret = WSASend(fd, reinterpret_cast<LPWSABUF>(const_cast<WindowsIOVec*>(iov)), count, &sent, 0, NULL, NULL);
187  if (ret == 0)
188  return sent;
189  return -1;
190 }
